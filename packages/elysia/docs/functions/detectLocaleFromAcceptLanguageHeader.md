## Returns

`string`

A locale string, which will be detected of **first** from `Accept-Language` header

## Example

```js
import { Elysia } from 'elysia'
import { intlify, detectLocaleFromAcceptLanguageHeader } from '@intlify/elysia'

new Elysia().use(
  intlify({
    messages: {
      en: {
        hello: 'Hello {name}!'
      },
      ja: {
        hello: '{name}'
      }
    },
    locale: detectLocaleFromAcceptLanguageHeader
  })
)
```
